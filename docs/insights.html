# Insight 2: Temporal Patterns – Rush Hour, Overnight, and Day of Week

# Aggregate mean tip by hour, day type, and rush/overnight 
temporal_summary <- ds |>
  mutate(
    day_type = if_else(pickup_dow %in% c("Saturday", "Sunday"), "Weekend", "Weekday"),
    time_period = case_when(
      overnight ~ "Overnight",
      rush_hour ~ "Rush Hour",
      TRUE ~ "Other"
    )
  ) |>
  group_by(pickup_hour, day_type, time_period) |>
  summarise(
    mean_tip = mean(tip_amount, na.rm = TRUE),
    median_tip = median(tip_amount, na.rm = TRUE),
    n = n(),
    se = sd(tip_amount, na.rm = TRUE) / sqrt(n),   # for error bands
    .groups = "drop"
  )

# 1. Heatmap
p3 <- ggplot(temporal_summary, aes(x = pickup_hour, y = day_type, fill = mean_tip)) +
  geom_tile(color = "white", linewidth = 0.3) +
  scale_fill_gradient2(
    low = "#f7fbff", mid = "#6baed6", high = "#08306b",
    midpoint = median(temporal_summary$mean_tip),
    name = "Mean Tip ($)"
  ) +
  scale_x_continuous(breaks = seq(0, 23, by = 2)) +
  labs(
    title = "Mean Tip Amount by Hour",
    subtitle = "Rush-hour and overnight premiums",
    x = "Pickup Hour (0–23)",
    y = NULL
  ) +
  theme_bw(base_size = 18) +
  theme(
    legend.position = "right",
    panel.grid.minor = element_blank()
  )

# 2. Line plot with confidence bands
p4 <- ggplot(temporal_summary, aes(x = pickup_hour, y = mean_tip, color = day_type, group = day_type)) +
  geom_line(linewidth = 1.1) +
  geom_ribbon(aes(ymin = mean_tip - 1.96 * se, ymax = mean_tip + 1.96 * se, fill = day_type),
              alpha = 0.15, color = NA) +
  geom_point(size = 2) +
  scale_color_manual(values = c("Weekday" = "#1f77b4", "Weekend" = "#ff7f0e")) +
  scale_fill_manual(values = c("Weekday" = "#1f77b4", "Weekend" = "#ff7f0e")) +
  scale_x_continuous(breaks = seq(0, 23, by = 2)) +
  labs(
    title = "Average Tip Amount Over the Day",
    subtitle = "Weekday vs Weekend",
    x = "Pickup Hour",
    y = "Mean Tip Amount ($)",
    color = "Day Type",
    fill = "Day Type"
  ) +
  theme_bw(base_size = 18) +
  theme(legend.position = "bottom")
